205class Note:
206    NOTES = ["C", "C#", "D", "Eb", "E", "F", "F#", "G", "Ab", "A", "Bb", "B"]
207    OCTAVES = list(range(11))
208    NOTES_IN_OCTAVE = len(NOTES)
209
210    def __init__(self) -> None:
211        self.pitch = ""
212        self.octave = 4
213        self.duration = Duration.Whole
214
215    def __swap_accidentals(note: str) -> str:
216        if note == "Db":
217            return "C#"
218        if note == "D#":
219            return "Eb"
220        if note == "E#":
221            return "F"
222        if note == "Gb":
223            return "F#"
224        if note == "G#":
225            return "Ab"
226        if note == "A#":
227            return "Bb"
228        if note == "B#":
229            return "C"
230        return note
231
232    def note_to_number(n) -> int:
233        """Returns numeric value of note as specified in midi format. Pauses (empty) notes are treated as -1"""
234
235        if n.pitch == "":
236            # We treat empty pitch as pause
237            return -1
238
239        note = Note.__swap_accidentals(n.pitch)
240        assert note in Note.NOTES, errors["notes"]
241        assert n.octave in Note.OCTAVES, errors["notes"]
242
243        note = Note.NOTES.index(note)
244        note += Note.NOTES_IN_OCTAVE * n.octave
245
246        assert 0 <= note <= 127, errors["notes"]
247        return note

250class Melody:
251    def __init__(self) -> None:
252        self.notes = []
253        self.track = 0
254        self.channel = 0
255        self.volume = 100
256        self.start_time = 0
257
258    def __get_time_offset(notes_duration: list[int], index: int) -> int:
259        return sum(notes_duration[:index])
260
261    def chords_to_notes(chord_progression: list[str], onlyFirst=False) -> list[str]:
262        """Encode list of chords as notes"""
263
264        array_of_notes = []
265        for chord in chord_progression:
266            if onlyFirst:
267                array_of_notes.append(chords.from_shorthand(chord)[0])
268            else:
269                array_of_notes.extend(chords.from_shorthand(chord))
270        return array_of_notes
271
272    def notes_to_midi(self, file: MIDIFile) -> None:
273        """Save the sequence of notes in the MIDI file"""
274
275        array_of_notes_numbers = []
276        notes_duration = [note.duration.value for note in self.notes]
277        for note in self.notes:
278            array_of_notes_numbers.append(Note.note_to_number(note))
279
280        for i, pitch in enumerate(array_of_notes_numbers):
281            if pitch > -1:
282                # we skip in adding pause as midi does not have such concept
283                file.addNote(
284                    self.track,
285                    self.channel,
286                    pitch,
287                    self.start_time + Melody.__get_time_offset(notes_duration, i),
288                    self.notes[i].duration.value,
289                    self.volume,
290                )

305def __note_to_number(note: str, octave: int) -> int:
306    note = __swap_accidentals(note)
307    assert note in NOTES, errors["notes"]
308    assert octave in OCTAVES, errors["notes"]
309
310    note = NOTES.index(note)
311    note += NOTES_IN_OCTAVE * octave
312
313    assert 0 <= note <= 127, errors["notes"]
314    return note

321def notes_to_midi(
322    file: MIDIFile,
323    array_of_notes: list[str],
324    notes_duration: list[float],
325    octave=4,
326    start_time=0,
327    track=0,
328    channel=0,
329    volume=100,
330) -> None:
331    """Save the sequence of notes in the MIDI file"""
332
333    assert len(array_of_notes) == len(notes_duration), errors["length"]
334
335    array_of_notes_numbers = []
336    for note in array_of_notes:
337        array_of_notes_numbers.append(__note_to_number(note, octave=octave))
338
339    for i, pitch in enumerate(array_of_notes_numbers):
340        file.addNote(
341            track,
342            channel,
343            pitch,
344            start_time + __get_time_offset(notes_duration, i),
345            notes_duration[i],
346            volume,
347        )
